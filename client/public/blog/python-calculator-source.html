Build a simple Python calculator (source code included)

This mini-project uses if/elif branching and simple input validation to build a console calculator. Copy the full code, run it in the online compiler, and then extend it with your own operations.

Calculator code

def calculate():
    print("Select operation: add, subtract, multiply, divide")
    op = input("Operation: ").strip().lower()
    a = float(input("First number: "))
    b = float(input("Second number: "))

    if op == "add":
        return a + b
    if op == "subtract":
        return a - b
    if op == "multiply":
        return a * b
    if op == "divide":
        if b == 0:
            return "Cannot divide by zero"
        return a / b
    return "Unknown operation"


if __name__ == "__main__":
    result = calculate()
    print("Result:", result)

Upgrade challenges
